用C语言求矩阵的秩

发表于 2019-02-17 | 分类于 C语言与矩阵 
矩阵是处理数据的一种方法，秩是矩阵的重要性质。因此，用C语言求解矩阵的秩可以帮我们更好的处理数据。首先，我们

先来看平时我们是如何求矩阵的秩的：选定矩阵的一行，再选另外一行减去k倍的第一行，通过改变k的值，将该行中的元素尽
可能多的化为0，最后通过行交换将矩阵化为行阶梯式，记录不全为零的行的个数即可得到该矩阵的秩。用C语言求矩阵的秩时，
我们也可使用此思路，即通过行变换，将原矩阵A中[0][0]下方元素全化为零，在讲A[1][1]下方元素全化为0，以此类推，知道
将矩阵化为行阶梯式。其代码实现如下：

    /*此处我们以一4*4矩阵为例*/  
   float arr[4][4];

   int main(void)
  {
   //定义所需的变量，具体意义在用到时详述
   int i=0,j=0,R=0,k=0,np=0,i_in=0,j_in=0,s=0,t=0;
   float get_arr=0,temp=0,mid=0,mid_arr=0;
  //输入数组 i表示行，j表示列
  for(i=0;i<4;i++)
 {
   for(j=0;j<4;j++)
  {
   printf("input the number of arr[%f][%f] ",i,j);
   scanf("%f",&get_arr);
   arr[i][j]=get_arr;
  }
 }
 //显示输入矩阵 
 printf("\n");
 for(i=0;i<4;i++)
{
 for(j=0;j<4;j++)
 {
  printf("%f ",arr[i][j]);
  }
  printf("\n");
 }


/*化为行阶梯式
  求解方法：我们以第一列为例，我们想将通过行变换使第一列最多只有一个非零数，且这个非零数位于第一行。为此，
            我们遍历这一列的元素，当出现一个非零数时，停止遍历，并将该行与第一行交换，使第一列第一行为非零
            数，再确定k值，将第一列二，三，四行元素化为零。如果第一列全为0，则跳至第二列，重复上述过程。
  变量意义：i表示行，j表示列，k表示检测到元素为非零数的次数，用于确定起始行
            i_in也表示行，用于循环中防止i的值被改变，j_in同理。np，用于确定
            一列是否全为0，如np=0则这一列为0
            mid和temp均为赋值的传递变量；
*/ 
for(j=k;j<4;j++)

{
for(i=k;i<4;i++)
{
if(arr[i][j]==0)
{
continue;
}
else
{
if(i>k)//说明该列检测的第一个元素为0
{
i_in=i;
j_in=j;
//交换两行数据，将非零行置顶
for(j_in=0;j_in<4;j_in++)
{
temp=arr[i_in][j_in];
arr[i_in][j_in]=arr[k][j_in];
arr[k][j_in]=temp;
}
}
//进行行计算
for(i_in=k;i_in<3;i_in++)
{
mid=arr[i_in+1][j]/arr[k][j];
for(j_in=j;j_in<4;j_in++)
{
mid_arr=arr[i_in+1][j_in]-mid*arr[k][j_in];
arr[i_in+1][j_in]=mid_arr;
}
}
k++;
np++;
}

  }

  if(np==0)
  {
   j++;
   np=0;
   }else
   {
    np=0;
}

 }
 //显示变换后矩阵 
   printf("\n");
   for(i=0;i<4;i++)
  {
   for(j=0;j<4;j++)
   {
     printf("%f ",arr[i][j]);
    }
     printf("\n");
   }

 //求秩，遍历每行，只要有非0元素即可 
   for(i=0;i<4;i++)
  {
   for(j=0;j<4;j++)
   {
    if(arr[i][j]!=0)
    {
    R++;
    break;
     }
    }
   }

   printf("R of arr is %d",R);

return 0;
}
